---
title: 'List Contacts'
api: 'GET /api/messaging/contacts/'
---

List all contacts with pagination and filtering.

<ParamField query="page" type="integer">
  Page number (default: 1)
</ParamField>

<ParamField query="page_size" type="integer">
  Items per page (default: 20, max: 100)
</ParamField>

<ParamField query="search" type="string">
  Search term
</ParamField>

<ParamField query="tags" type="string">
  Filter by tags (comma-separated)
</ParamField>

## Success Response (200)

<ResponseField name="count" type="integer">
  Total number of contacts
</ResponseField>

<ResponseField name="next" type="string">
  URL to next page (nullable)
</ResponseField>

<ResponseField name="previous" type="string">
  URL to previous page (nullable)
</ResponseField>

<ResponseField name="results" type="array">
  Array of contact objects
</ResponseField>
